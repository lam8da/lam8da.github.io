\begin{algorithmic}[1] % The number tells where the line numbering should start
  \For{\textbf{each} $CC_i\in CC$}
    \For{\textbf{each} LR(1) item $I\in CC_i$}
      \If{$I$ is $[A\to\beta\cdot c\gamma a]$ and $goto(CC_i,c)=CC_j$}
        \State $Action[i,c] \gets $"shift $j$"
      \ElsIf{$I$ is $[A\to\beta\cdot,a]$}
        \State $Action[i,a] \gets $"reduce $A\to\beta$"
      \ElsIf{$I$ is $[S'\to S\cdot,eof]$}
        \State $Action[i,eof] \gets $"accept"
      \EndIf
    \EndFor
    \For{\textbf{each} $n\in NT$}
      \If{$goto(CC_i,n)=CC_j$}
        \State $Goto[i,n] \gets j$
      \EndIf
    \EndFor
  \EndFor
\end{algorithmic}
