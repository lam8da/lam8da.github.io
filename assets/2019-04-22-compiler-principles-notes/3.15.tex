\begin{document}

\begin{algorithmic}[1] % The number tells where the line numbering should start
  \State push $\$$
  \State push start state $s_0$
  \State $word \gets NextWord()$
  \LoopForever
    \State $state \gets $ top of stack
    \If{$Action[state,word]=$ "reduce $A\to\beta_1\beta_2\cdots\beta_k$"}
      \LineComment{因为每次都push两个元素：一个是符号（终结符或非终结符），另一个是语法分析的状态}
      \State pop $2k$ symbols
      \State $state \gets $ top of stack
      \State push $A$
      \State push $Goto[state,A]$
    \ElsIf{$Action[state,word]=$ "shift $s_i$"}
      \State push $word$
      \State push $s_i$
      \State $word \gets NextWord()$
    \ElsIf{$Action[state,word]=$ "accept"}
      \State \textbf{break}
    \Else
      \State $Fail()$
    \EndIf
  \EndLoopForever
  \State report success
\end{algorithmic}
