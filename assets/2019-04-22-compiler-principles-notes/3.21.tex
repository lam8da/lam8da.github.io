\begin{algorithmic}[1] % The number tells where the line numbering should start
  \Function{$goto$}{$s,x$}  \LeftComment{$s$是一个LR(1)项集，$s$为一个语法符号}
    \State $moved \gets \emptyset$
    \For{\textbf{each} LR(1) item $i\in s$}
      \If{the form of $i$ is $[\alpha\to\beta\cdot x\delta,a]$}
        \State $moved \gets moved\cup\{[\alpha\to\beta x\cdot\delta,a]\}$
      \EndIf
    \EndFor
    \StateReturn $closure(moved)$
  \EndFunction
\end{algorithmic}
